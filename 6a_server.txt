from flask import Flask, request, jsonify, render_template
import os
import sys

# Add EmotionDetection package path to system environment
sys.path.insert(0, os.path.abspath(os.path.dirname(__file__)))
from EmotionDetection import load_emotion_model, emotion_detector

# Initialize Flask application
app = Flask(__name__)

# Load emotion detection model once at startup
model = load_emotion_model(model_path="text_classification_model.tflite")

# Define home route (web interface)
@app.route('/')
def home():
    return render_template('index.html')

# Define API route for emotion detection (POST method)
@app.route('/detect-emotion', methods=['POST'])
def detect_emotion_api():
    try:
        # Get input text from request
        data = request.get_json()
        if not data or 'text' not in data:
            return jsonify({
                "error": "Invalid request: 'text' field is required",
                "status": "failed"
            }), 400
        
        input_text = data['text'].strip()
        if not input_text:
            return jsonify({
                "error": "Input text cannot be empty",
                "status": "failed"
            }), 400
        
        # Run emotion detection
        result = emotion_detector(input_text, model)
        
        # Return formatted response
        return jsonify({
            "input_text": result["input_text"],
            "predictions": result["predictions"],
            "dominant_emotion": result["dominant_emotion"],
            "dominant_score": result["dominant_score"],
            "status": "success"
        }), 200
    
    except TypeError as e:
        return jsonify({
            "error": f"Invalid input type: {str(e)}",
            "status": "failed"
        }), 400
    
    except Exception as e:
        return jsonify({
            "error": f"Server error: {str(e)}",
            "status": "failed"
        }), 500

# Define health check route
@app.route('/health', methods=['GET'])
def health_check():
    return jsonify({
        "status": "healthy",
        "service": "EmotionDetection API",
        "model_loaded": model is not None
    }), 200

# Run Flask server
if __name__ == "__main__":
    # Create templates directory if not exists (for HTML interface)
    if not os.path.exists('templates'):
        os.makedirs('templates')
    
    # Write basic index.html to templates directory
    with open('templates/index.html', 'w') as f:
        f.write("""
<!DOCTYPE html>
<html>
<head>
    <title>Emotion Detection</title>
    <style>
        body { font-family: Arial, sans-serif; max-width: 800px; margin: 2rem auto; padding: 0 1rem; }
        .container { border: 1px solid #ddd; padding: 2rem; border-radius: 8px; }
        input { width: 100%; padding: 0.8rem; margin: 1rem 0; border: 1px solid #ddd; border-radius: 4px; }
        button { padding: 0.8rem 2rem; background: #007bff; color: white; border: none; border-radius: 4px; cursor: pointer; }
        #result { margin-top: 2rem; padding: 1rem; border: 1px solid #ddd; border-radius: 4px; }
    </style>
</head>
<body>
    <div class="container">
        <h1>Emotion Detection Tool</h1>
        <input type="text" id="textInput" placeholder="Enter text to detect emotion...">
        <button onclick="detectEmotion()">Detect Emotion</button>
        <div id="result"></div>
    </div>

    <script>
        async function detectEmotion() {
            const text = document.getElementById('textInput').value;
            const resultDiv = document.getElementById('result');
            
            if (!text) {
                resultDiv.innerHTML = "<p style='color: red;'>Please enter text!</p>";
                return;
            }

            try {
                const response = await fetch('/detect-emotion', {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({ text: text })
                });
                const data = await response.json();
                
                if (data.status === 'success') {
                    let predictions = '';
                    data.predictions.forEach(pred => {
                        predictions += `<p>${pred.category}: ${pred.score.toFixed(5)}</p>`;
                    });
                    resultDiv.innerHTML = `
                        <h3>Result</h3>
                        <p><strong>Input Text:</strong> ${data.input_text}</p>
                        <p><strong>Dominant Emotion:</strong> ${data.dominant_emotion} (Score: ${data.dominant_score.toFixed(5)})</p>
                        <h4>All Predictions:</h4>
                        ${predictions}
                    `;
                } else {
                    resultDiv.innerHTML = `<p style='color: red;'>Error: ${data.error}</p>`;
                }
            } catch (err) {
                resultDiv.innerHTML = `<p style='color: red;'>Server connection error: ${err}</p>`;
            }
        }
    </script>
</body>
</html>
        """)
    
    # Start server (debug mode for development, disable in production)
    app.run(host='0.0.0.0', port=5000, debug=True)
